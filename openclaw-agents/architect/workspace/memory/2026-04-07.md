```python
def _get_best_practices(self, conn, project_id):
    cursor = conn.execute("""
        SELECT category, practice, evidence, avg_score
        FROM best_practices
        WHERE (project_id = ? OR project_id IS NULL)
        AND avg_score >= 90
        ORDER BY avg_score DESC, category
        LIMIT 20
    """, (project_id,))
    rows = cursor.fetchall()
    if not rows:
        return None
    
    parts = []
    for row in rows:
        parts.append(f"- [{row['category']}] {row['practice']}")
        if row['evidence']:
            parts.append(f"  示例：{row['evidence'][:100]}")
    
    return "\n".join(parts)
```

#### 4.2 自动合并脚本

```python
# shared/tools/pattern_merger.py

def promote_high_frequency_patterns(threshold=3):
    """将高频 learned_patterns 提升为 anti_patterns"""
    conn = get_connection()
    
    # 查找高频问题
    cursor = conn.execute("""
        SELECT DISTINCT category, pattern, COUNT(*) as freq
        FROM learned_patterns
        WHERE frequency >= ?
        GROUP BY category, pattern
    """, (threshold,))
    
    for row in cursor.fetchall():
        # 检查是否已存在
        existing = conn.execute(
            "SELECT id FROM anti_patterns WHERE pattern = ?",
            (row['pattern'],)
        ).fetchone()
        
        if not existing:
            # 新增到 anti_patterns
            code = f"LP{row['id']:03d}"
            conn.execute("""
                INSERT INTO anti_patterns (code, category, pattern, description, severity)
                VALUES (?, ?, ?, ?, 'medium')
            """, (code, row['category'], row['pattern'], f"从项目经验学习：{row['pattern']}"))
    
    conn.commit()
```
